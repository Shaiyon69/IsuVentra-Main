import 'dart:convert';
import 'package:flutter/foundation.dart';
import '../models/participation_model.dart';
import 'api_service.dart';

class ParticipationService {
  final ApiService _apiService = ApiService();

  Future<List<Participation>> getParticipations() async {
    try {
      final response = await _apiService.get('/participation');

      if (response.statusCode == 200) {
        final List<dynamic> data = jsonDecode(response.body);
        return data.map((json) => Participation.fromJson(json)).toList();
      } else {
        debugPrint('Get participations failed: ${response.statusCode}');
        return [];
      }
    } catch (e) {
      debugPrint('Get participations error: $e');
      return [];
    }
  }

  Future<Participation?> getParticipation(int id) async {
    try {
      final response = await _apiService.get('/participation/$id');

      if (response.statusCode == 200) {
        final data = jsonDecode(response.body);
        return Participation.fromJson(data);
      } else {
        debugPrint('Get participation failed: ${response.statusCode}');
        return null;
      }
    } catch (e) {
      debugPrint('Get participation error: $e');
      return null;
    }
  }
}
